    set exts=%%i\*.d88 %%i\*.t88 %%i\*.cmt
    set "curr=%%i"
    
    for %%j in (!exts!) do (

        if exist "!temp!" (
            for /f "tokens=*" %%k in (!temp!) do (
                copy %%k to "!non_disk!"

            del "!temp!"

        )

        rem parse disk or Disk from file name
        for /f "tokens=2 delims=(" %%k in ("%%j") do (
            set "!wrd_tail=%%k"

            set "wrd=%%k"
